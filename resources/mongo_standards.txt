- database name: slow_cooker
- collections: rpi_address, temperature, cook_time, lid_status

- Note: All collections will have a date field, but this will be handled by the server.

- rpi_address = {
    "address": IP address,
}

- temperature = {
    "type": [probe/vessel]
    "temperature": probe->[140-180F in 5 degree increments]/vessel->[keep_warm/medium/high]
    "measurement": [F/C]
}

- cook_time = {
    "start_time": UTC time
}

- lid_status = {
    "status": [secure/unsecure]
}
